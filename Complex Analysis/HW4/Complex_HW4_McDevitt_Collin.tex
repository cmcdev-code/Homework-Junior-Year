\begin{tcolorbox}
  \textbf{Problem 1.}
    Compute the fractional linear transformation determined by the correspondence:
    \[
        (0,1,\infty)\mapsto (1,1+i,2)
    \]
  
\end{tcolorbox}

\[f(z)=\frac{z(2-2i)-2}{z(1-i)-2}\]

Then $f(0)=1$, $f(1)=\frac{2i}{1+i}=\frac{2i(1-i)}{|1+i|}=i(1-i)=i+1$, and $f(\infty)=2$.
